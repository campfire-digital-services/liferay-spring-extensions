017    package au.com.permeance.liferay.spring;
018    
019    import com.liferay.portal.kernel.bean.BeanLocator;
020    import com.liferay.portal.kernel.bean.BeanLocatorException;
021    import com.liferay.portal.kernel.log.Log;
022    
023    import java.io.IOException;
024    import java.util.Collection;
025    import java.util.LinkedList;
026    import java.util.List;
027    
028    import org.springframework.beans.factory.config.SingletonBeanRegistry;
029    import org.springframework.context.ApplicationContext;
030    import org.springframework.context.ConfigurableApplicationContext;
031    import org.springframework.context.support.ApplicationObjectSupport;
032    import org.springframework.core.type.classreading.MetadataReader;
033    import org.springframework.core.type.filter.TypeFilter;
034    
035    import static com.liferay.portal.kernel.log.LogFactoryUtil.getLog;
036    
037    import static java.lang.Class.forName;
038    import static java.lang.String.format;
039    
040    /**
041     * This class provides a bean which registers existing liferay beans obtained from liferay's {@link BeanLocator} in the
042     * Spring {@link ApplicationContext} in which the instance of this class is defined. Bean inclusions and exclusions can
043     * be specified using the {@link #addIncludeFilter(TypeFilter)} and {@link #addExcludeFilter(TypeFilter)} methods.
044     * Exclusions take priority of inclusions.
045     */
046    public class BeanLocatorDefinitionCopier extends ApplicationObjectSupport {
047    
048        /**
049         * Logger for this class.
050         */
051        private static final Log LOG = getLog(BeanLocatorDefinitionCopier.class);
052    
053        /**
054         * Stores the list of bean exclusions.
055         */
056        private final transient List<TypeFilter> excludeFilters = new LinkedList<TypeFilter>();
057    
058        /**
059         * Stores the list of bean inclusions.
060         */
061        private final transient List<TypeFilter> includeFilters = new LinkedList<TypeFilter>();
062    
063        /**
064         * Stores the liferay bean locator to copy beans from.
065         */
066        private final transient BeanLocator beanLocator;
067    
068        /**
069         * Creates a new instance based on the supplied bean locator.
070         *
071         * @param beanLocator the liferay bean locator to copy beans from.
072         */
073        public BeanLocatorDefinitionCopier(final BeanLocator beanLocator) {
074            super();
075            LOG.debug(format("Storing bean locator: %s", beanLocator));
076            this.beanLocator = beanLocator;
077        }
078    
079        /**
080         * Adds an inclusion filter.
081         *
082         * @param filter the filter which beans must match to be copied.
083         */
084        public final void addIncludeFilter(final TypeFilter filter) {
085            includeFilters.add(filter);
086        }
087    
088        /**
089         * Adds an exclusion filter.
090         *
091         * @param filter the filter which beans must not match to be copied.
092         */
093        public final void addExcludeFilter(final TypeFilter filter) {
094            excludeFilters.add(filter);
095        }
096    
097        /**
098         * {@inheritDoc}
099         *
100         * @return always returns {@link ConfigurableApplicationContext}.
101         */
102        @Override
103        protected final Class requiredContextClass() {
104            return ConfigurableApplicationContext.class;
105        }
106    
107        /**
108         * Initialises the supplied application context based on the {@link BeanLocator} associated with this instance by
109         * iterating the liferay registered beans (via {@link BeanLocator#getNames()}).
110         *
111         * @param context the {@link ConfigurableApplicationContext} in which beans should be registered (by calling
112         *                {@link ConfigurableApplicationContext#getBeanFactory()}).
113         */
114        @Override
115        protected final void initApplicationContext(final ApplicationContext context) {
116            LOG.info(format("Copying bean definitions from %s to %s", beanLocator, context));
117    
118            final ConfigurableApplicationContext configurableApplicationContext = (ConfigurableApplicationContext) context;
119            @SuppressWarnings("PMD.DataflowAnomalyAnalysis")
120            final SingletonBeanRegistry singletonBeanRegistry = configurableApplicationContext.getBeanFactory();
121    
122            final String[] names = beanLocator.getNames();
123            for (String name : names) {
124                LOG.debug(format("Processing bean locator bean named: %s", name));
125    
126                if (!isAcceptable(name)) {
127                    LOG.debug(format("Skipping bean %s", name));
128                    continue;
129                }
130    
131                final Object bean = safeLocate(name);
132                if (bean == null) {
133                    LOG.warn(format("Skipping bean %s (bean locator couldn't acquire a valid instance)", name));
134                    continue;
135                }
136    
137                LOG.info(format("Copying bean definition %s", name));
138                singletonBeanRegistry.registerSingleton(name, bean);
139            }
140        }
141    
142        /**
143         * Calls {@link BeanLocator#locate(String)} on the {@link #beanLocator} associated with this instance, catching and
144         * logging any {@link BeanLocatorException} which is thrown.
145         *
146         * @param name the name of the bean to locate.
147         *
148         * @return the instance of the specified bean if it exists, or {@code null} if a {@link BeanLocatorException} is
149         *         thrown.
150         */
151        protected final Object safeLocate(final String name) {
152            LOG.debug(format("Fetching bean from bean locator: %s", name));
153            try {
154                return beanLocator.locate(name);
155            }
156            catch (final BeanLocatorException e) {
157                LOG.warn(format("Error fetching bean from bean locator: %s - skipping", name), e);
158            }
159            return null;
160        }
161    
162        /**
163         * Determines whether the class identified by the supplied string is acceptable for copying by this instance.
164         * Acceptance is based on (in order):
165         * <ol>
166         * <li>The string identifying a known type (if the type is unknown, it is unacceptable)</li>
167         * <li>Which does not match an exclude filter (if the type is excluded, it is unacceptable)</li>
168         * <li>Which matches any defined include filters (if the type is included, it is acceptable)</li>
169         * <li>Or if no include filter is defined, it is acceptable</li>
170         * </ol>
171         *
172         * @param type the type of class to match.
173         *
174         * @return {@code true} if the type matches the acceptance criteria for this class, {@code false} otherwise.
175         */
176        protected final boolean isAcceptable(final String type) {
177            final Class<?> clazz = safeForName(type);
178    
179            if (clazz == null) {
180                LOG.debug(format("Skipping unknown type %s", type));
181                return false;
182            }
183    
184            if (matches(excludeFilters, clazz)) {
185                LOG.debug(format("Refusing excluded type %s", type));
186                return false;
187            }
188    
189            if (matches(includeFilters, clazz)) {
190                LOG.debug(format("Accepting included type %s", type));
191                return true;
192            }
193    
194            return includeFilters.isEmpty();
195        }
196    
197        /**
198         * Checks whether the supplied class matches any of the supplied type filters, returning {@code true} if it does,
199         * {@code false} otherwise.
200         *
201         * @param filters the filters to match against.
202         * @param clazz   the class to match.
203         *
204         * @return {@code true} if any of the filters match, {@code false} otherwise.
205         */
206        protected final boolean matches(final Collection<? extends TypeFilter> filters, final Class<?> clazz) {
207            @SuppressWarnings("PMD.DataflowAnomalyAnalysis")
208            final MetadataReader metadataReader = new SimpleMetadataReader(clazz);
209            for (TypeFilter filter : filters) {
210                final boolean match = safeMatch(filter, metadataReader);
211                if (match) {
212                    return true;
213                }
214            }
215            return false;
216        }
217    
218    
219        /**
220         * Wraps a call to
221         * {@link TypeFilter#match(MetadataReader, org.springframework.core.type.classreading.MetadataReaderFactory)}
222         * so that any {@link IOException} is logged rather than propagating up the call stack. Returns the result of the
223         * match if successful, {@code false} otherwise.
224         *
225         * @param filter         the filter to match against.
226         * @param metadataReader the metadata reader to match with.
227         *
228         * @return true if
229         *         {@link TypeFilter#match(MetadataReader, org.springframework.core.type.classreading.MetadataReaderFactory)}
230         *         returns it, {@code false} otherwise.
231         */
232        protected final boolean safeMatch(final TypeFilter filter, final MetadataReader metadataReader) {
233            try {
234                return filter.match(metadataReader, null);
235            }
236            catch (final IOException e) {
237                LOG.warn(format("Error checking for filter %s match against reader %s", filter, metadataReader));
238            }
239            return false;
240        }
241    
242        /**
243         * Wraps a call to {@link Class#forName(String)} so that any {@link ClassNotFoundException} is logged rather than
244         * propagating back up the call stack. Returns the specified class if successful, or {@code null} otherwise.
245         *
246         * @param className the name of the class to resolve.
247         *
248         * @return the requested class, if available, or {@code null}.
249         */
250        protected final Class<?> safeForName(final String className) {
251            try {
252                return forName(className);
253            }
254            catch (final ClassNotFoundException e) {
255                LOG.debug(format("Unknown class %s", className));
256            }
257            return null;
258        }
