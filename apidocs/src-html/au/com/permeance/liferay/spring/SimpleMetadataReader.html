017    package au.com.permeance.liferay.spring;
018    
019    import org.springframework.core.io.Resource;
020    import org.springframework.core.type.StandardAnnotationMetadata;
021    import org.springframework.core.type.classreading.MetadataReader;
022    
023    /**
024     * This class provides an implementation of the {@link MetadataReader} interface which reads from an already
025     * materialised class (rather than a resource or stream).
026     */
027    public class SimpleMetadataReader implements MetadataReader {
028    
029        /**
030         * Stores the metadata for the class supplied in the constructor.
031         */
032        private final transient StandardAnnotationMetadata standardAnnotationMetadata;
033    
034        /**
035         * Creates a new instance, based on the supplied class.
036         *
037         * @param clazz the class to base this instance on.
038         */
039        public SimpleMetadataReader(final Class<?> clazz) {
040            standardAnnotationMetadata = new StandardAnnotationMetadata(clazz);
041        }
042    
043        /**
044         * {@inheritDoc}
045         *
046         * @return always returns {@code null}.
047         */
048        @Override
049        public final Resource getResource() {
050            return null;
051        }
052    
053        /**
054         * {@inheritDoc}
055         *
056         * @return returns {@link #standardAnnotationMetadata}
057         */
058        @Override
059        public final StandardAnnotationMetadata getClassMetadata() {
060            return standardAnnotationMetadata;
061        }
062    
063        /**
064         * {@inheritDoc}
065         *
066         * @return returns {@link #standardAnnotationMetadata}
067         */
068        @Override
069        public final StandardAnnotationMetadata getAnnotationMetadata() {
070            return standardAnnotationMetadata;
071        }
072    
073    }
